-- Basic SQL Keywords

SELECT
-- Retrieves data from a table.
-- Example: SELECT name FROM users;

FROM
-- Specifies the table to pull data from.
-- Example: SELECT * FROM products;

AS
-- Renames a column or table temporarily (alias).
-- Example: SELECT price AS product_price FROM items;

DISTINCT
-- Removes duplicate results.
-- Example: SELECT DISTINCT country FROM customers;

WHERE
-- Filters records based on a condition.
-- Example: SELECT * FROM orders WHERE quantity > 5;

ORDER BY
-- Sorts results by one or more columns.
-- Example: ORDER BY release_date;

DESC
-- Sorts in descending order (big to small).
-- Example: ORDER BY score DESC;

ASC
-- Sorts in ascending order (small to big, default).
-- Example: ORDER BY name ASC;

LIMIT
-- Limits the number of rows returned.
-- Example: LIMIT 5;

-- Pattern Matching & Null Checks

LIKE
-- Searches for a pattern in text.
-- Example: WHERE name LIKE 'A%' (starts with A);

_
-- Wildcard for a single character (used with LIKE).
-- Example: WHERE code LIKE 'A_1' (matches A01, AB1, etc.);

%
-- Wildcard for zero or more characters (used with LIKE).
-- Example: WHERE title LIKE '%game%' (contains "game");

IS NULL
-- Checks if a value is null (missing/empty).
-- Example: WHERE description IS NULL;

IS NOT NULL
-- Checks if a value is not null.
-- Example: WHERE rating IS NOT NULL;

-- Filtering

BETWEEN
-- Filters values within a range (inclusive).
-- Example: WHERE price BETWEEN 10 AND 20;

-- Conditional Logic

CASE ... END
-- Creates conditional logic (like if-else).
-- Example:
-- SELECT

--   CASE 
--     WHEN rating >= 90 THEN 'Excellent'
--     WHEN rating >= 70 THEN 'Good'
--     ELSE 'Average'
--   END AS quality
-- FROM games;

-- Table and Data Management

CREATE TABLE
-- Creates a new table with specified columns.
-- Example:
-- CREATE TABLE users (
--   id INTEGER PRIMARY KEY,
--   name TEXT,
--   signup_date DATE DEFAULT CURRENT_DATE
-- );

ALTER TABLE
-- Modifies an existing table (add/remove columns, etc.).
-- Example: ALTER TABLE users ADD COLUMN email TEXT;

ADD COLUMN
-- Adds a new column to an existing table.
-- Example: ALTER TABLE users ADD COLUMN is_active INTEGER;

INSERT INTO
-- Adds new rows of data to a table.
-- Example: INSERT INTO users (name, signup_date) VALUES ('Alice', '2023-01-01');

UPDATE
-- Modifies existing records in a table.
-- Example: UPDATE users SET name = 'Bob' WHERE id = 1;

DELETE FROM
-- Deletes rows from a table.
-- Example: DELETE FROM users WHERE id = 3;

PRIMARY KEY
-- Uniquely identifies each row in a table. Usually used with an INTEGER column.
-- Example: id INTEGER PRIMARY KEY

DEFAULT
-- Assigns a default value when none is provided during insertion.
-- Example: signup_date DATE DEFAULT CURRENT_DATE

-- Data Types

INTEGER
-- Stores whole numbers.
-- Example: age INTEGER

TEXT
-- Stores text strings.
-- Example: name TEXT

DATE
-- Stores calendar dates.
-- Example: signup_date DATE

REAL
-- Stores decimal numbers.
-- Example: price REAL

-- Aggregate Functions

COUNT()
-- Returns the number of rows that match a condition.
-- Example: SELECT COUNT(*) FROM users;

SUM()
-- Returns the total of a numeric column.
-- Example: SELECT SUM(price) FROM orders;

MAX()
-- Returns the highest value in a column.
-- Example: SELECT MAX(score) FROM games;

MIN()
-- Returns the lowest value in a column.
-- Example: SELECT MIN(score) FROM games;

AVG()
-- Returns the average of a numeric column.
-- Example: SELECT AVG(price) FROM products;

ROUND()
-- Rounds a numeric value to a specified number of decimal places.
-- Example: SELECT ROUND(price, 2) FROM products;
